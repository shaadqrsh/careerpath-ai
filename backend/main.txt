def sanitize_input(text: str) -> str:
    if not isinstance(text, str):
        return str(text)
    return re.sub(r'[^\w\s.,!?-]', '', text).strip()

def sanitize_quiz_answers(answers: List[Dict[str, Any]]) -> str:
    clean_text = []
    for a in answers:
        q_id = a.get("questionId", "Unknown")
        ans = sanitize_input(str(a.get("answer", "")))
        clean_text.append(f"Q{q_id}: {ans}")
    return "; ".join(clean_text)

def get_user_client(token: str) -> Client:
    client = create_client(SUPABASE_URL, SUPABASE_KEY)
    client.postgrest.auth(token)
    return client

async def verify_token(authorization: str = Header(None)):
    if not authorization or not authorization.startswith("Bearer "):
        raise HTTPException(status_code=401, detail="Missing or invalid token")
    
    token = authorization.split(" ")[1]
    
    try:
        user = supabase.auth.get_user(token)
        if not user:
            raise HTTPException(status_code=401, detail="Invalid token")
        return user.user
    except Exception as e:
        logger.error(f"Auth error: {e}")
        raise HTTPException(status_code=401, detail="Authentication failed")

def check_quota(client: Client, user_id: str, count_col: str, date_col: str, limit: int) -> int:
    try:
        res = client.table("profiles").select(f"{count_col}, {date_col}").eq("id", user_id).single().execute()
        if not res.data: return 0 
        
        data = res.data
        count = data.get(count_col, 0)
        last_date_str = data.get(date_col)
        
        now = datetime.now(timezone.utc)
        
        if last_date_str:
            try:
                last_date = datetime.fromisoformat(last_date_str.replace('Z', '+00:00'))
            except ValueError:
                last_date = now
        else:
            last_date = now
        
        if now.date() > last_date.date():
            try:
                client.table("profiles").update({
                    count_col: 0,
                    date_col: now.isoformat()
                }).eq("id", user_id).execute()
                count = 0
            except Exception as update_error:
                logger.error(f"Failed to reset daily quota: {update_error}")
        
        if count >= limit:
             raise HTTPException(status_code=429, detail=f"Daily limit reached ({limit}). Please try again tomorrow.")
        
        return count
    except HTTPException as he:
        raise he
    except Exception as e:
        logger.error(f"Quota check error: {e}")
        raise HTTPException(status_code=500, detail="Could not verify quota.")

def update_quota(client: Client, user_id: str, count_col: str, new_count: int):
    try:
        client.table("profiles").update({
            count_col: new_count,
            "last_image_generation_date" if "image" in count_col else "last_career_generation_date": datetime.now(timezone.utc).isoformat()
        }).eq("id", user_id).execute()
    except Exception as e:
        logger.error(f"Failed to update quota: {e}")

@app.post("/api/auth/signup")
async def signup(creds: UserLogin):
    try:
        res = supabase.auth.sign_up({
            "email": creds.email,
            "password": creds.password,
        })
        if res.user and (res.user.identities is not None) and len(res.user.identities) == 0:
            raise Exception("User already registered")
        return res
    except Exception as e:
        msg = str(e)
        if "User already registered" in msg:
             raise HTTPException(status_code=400, detail="User already registered")
        raise HTTPException(status_code=400, detail=msg)

@app.post("/api/profile")
async def upsert_profile(profile: UserProfile, authorization: str = Header(None), user = Depends(verify_token)):
    if profile.id != user.id:
        raise HTTPException(status_code=403, detail="Cannot update other users")
    
    token = authorization.split(" ")[1]
    client = get_user_client(token)

    try:
        data = profile.model_dump(by_alias=False)
        keys_to_remove = [
            'daily_generations_count', 'last_generation_date', 
            'daily_image_generations_count', 'last_image_generation_date',
            'daily_career_generations_count', 'last_career_generation_date'
        ]
        for k in keys_to_remove:
            if k in data: del data[k]
        
        res = client.table("profiles").upsert(data).execute()
        return {"status": "success"}
    except Exception as e:
        logger.error(f"Upsert Profile Error: {e}")
        raise HTTPException(status_code=500, detail=str(e))

@app.post("/api/generate-domain")
async def generate_domain(req: GenerateDomainRequest, authorization: str = Header(None), user = Depends(verify_token)):
    token = authorization.split(" ")[1]
    client = get_user_client(token)
    
    current_count = check_quota(client, user.id, "daily_career_generations_count", "last_career_generation_date", DAILY_CAREER_LIMIT)
    
    try:
        clean_answers = sanitize_quiz_answers(req.quiz_answers)

        prompt = f"""
        Analyze these quiz answers from a General Career Assessment.
        Identify the most suitable broad domain for this user.
        Context: {clean_answers}
        Options: 'science', 'commerce', 'arts'.
        Output strictly JSON: {{ "suggested_domain": "science" }}
        """
        model = genai.GenerativeModel(TEXT_MODEL_NAME)
        response = await model.generate_content_async(prompt, generation_config={"response_mime_type": "application/json"})
        
        return json.loads(response.text)
    except Exception as e:
        logger.error(f"Domain Gen Error: {e}")
        return {"suggested_domain": "general"}

@app.post("/api/generate-career")
async def generate_career(req: GenerateCareerRequest, authorization: str = Header(None), user = Depends(verify_token)):
    token = authorization.split(" ")[1]
    client = get_user_client(token)

    current_count = check_quota(client, user.id, "daily_career_generations_count", "last_career_generation_date", DAILY_CAREER_LIMIT)

    try:
        user_p = req.user_profile
        
        clean_answers = sanitize_quiz_answers(req.quiz_answers)
        safe_full_name = sanitize_input(user_p.full_name)
        safe_specialization = sanitize_input(user_p.specialization)

        prompt = f"""
        Act as an expert Career Counselor.
        Profile: {safe_full_name}, {user_p.age}y/o, {user_p.education_level} in {safe_specialization}.
        Residence: {user_p.residence_country}, Target: {user_p.preferred_work_country}.
        Traits: {clean_answers}.
        
        Task:
        1. Identify {CAREERS_PER_GENERATION} personalized career paths.
        2. STRICT ELIGIBILITY CHECK: If user is Arts, no Science.
        3. Provide only high-level details. Do NOT generate roadmaps or pivot analysis yet.
        
        OUTPUT FORMAT: STRICT JSON.
        {{
          "recommendations": [
            {{
              "id": "uuid-placeholder",
              "title": "Job Title",
              "matchScore": 85,
              "summary": "2-3 sentence summary of why this fits.",
              "salaryRange": "$50k - $80k",
              "growth": "High",
              "tags": ["Tag1", "Tag2"],
              "roadmap": [],
              "dayInLifePrompts": []
            }}
          ]
        }}
        """
        model = genai.GenerativeModel(TEXT_MODEL_NAME)
        response = await model.generate_content_async(prompt, generation_config={"response_mime_type": "application/json"})
        result_json = json.loads(response.text)
        
        for rec in result_json.get("recommendations", []):
            if not rec.get("id"): rec["id"] = str(uuid.uuid4())
            rec["roadmap"] = []
            rec["dayInLifePrompts"] = []
            
        update_quota(client, user.id, "daily_career_generations_count", current_count + 1)
        
        return result_json
    except Exception as e:
        logger.error(f"Career Gen Error: {e}")
        raise HTTPException(status_code=500, detail="AI generation failed")

@app.post("/api/generate-images")
async def generate_images(req: GenerateImagesRequest, authorization: str = Header(None), user = Depends(verify_token)):
    token = authorization.split(" ")[1]
    client = get_user_client(token)

    current_count = check_quota(client, user.id, "daily_image_generations_count", "last_image_generation_date", DAILY_IMAGE_LIMIT)

    try:
        safe_career_title = sanitize_input(req.career_title)

        subject_desc = "a professional"
        if req.user_context:
            u = req.user_context
            age = req.future_age or (u.age + 5)
            subject_desc = f"a {age} year old {u.gender} {u.residence_country} national"

        async def generate_single_image(prompt_text):
            safe_prompt = sanitize_input(prompt_text)
            full_prompt = f"Cinematic shot of {subject_desc} working as a {safe_career_title}. Scene: {safe_prompt}. Style: Photorealistic, 4k, dramatic lighting, inspiring, prestigious atmosphere."
            
            try:
                model = genai.GenerativeModel(IMAGE_MODEL_NAME)
                response = await model.generate_content_async(full_prompt)
                
                if response.candidates and response.candidates[0].content.parts:
                    for part in response.candidates[0].content.parts:
                        if part.inline_data:
                            b64_data = base64.b64encode(part.inline_data.data).decode('utf-8')
                            return f"data:{part.inline_data.mime_type};base64,{b64_data}"
            except Exception as e:
                pass

            try:
                safe_prompt_encoded = full_prompt[:300] 
                encoded_prompt = urllib.parse.quote(safe_prompt_encoded)
                seed = uuid.uuid4().int % 10000
                image_url = f"https://image.pollinations.ai/prompt/{encoded_prompt}?width=1280&height=720&seed={seed}&nologo=true"
                
                headers = {'User-Agent': 'CareerPathAI/1.0'}
                async with httpx.AsyncClient(follow_redirects=True, timeout=30.0) as client:
                    resp = await client.get(image_url, headers=headers)
                    if resp.status_code == 200:
                        b64_img = base64.b64encode(resp.content).decode('utf-8')
                        return f"data:image/jpeg;base64,{b64_img}"
            except Exception as e:
                pass

            return None

        generated_images = []
        for i, p in enumerate(req.prompts):
            if i > 0:
                await asyncio.sleep(1.5)
            
            img = await generate_single_image(p)
            generated_images.append(img)
        
        if any(img is not None for img in generated_images):
            update_quota(client, user.id, "daily_image_generations_count", current_count + 1)

        return {"images": generated_images}
        
    except Exception as e:
        return {"images": [None for _ in req.prompts]}

@app.post("/api/saved-careers")
async def save_career(career: CareerRecommendation, authorization: str = Header(None), user = Depends(verify_token)):
    token = authorization.split(" ")[1]
    client = get_user_client(token)
    try:
        data = career.model_dump(by_alias=False)
        data['user_id'] = user.id
        data['career_uid'] = career.id
        if 'id' in data: del data['id']

        if 'slide_images' in data and data['slide_images']:
            cleaned_images = []
            for img in data['slide_images']:
                if img and len(img) > 2000:
                    cleaned_images.append(None)
                else:
                    cleaned_images.append(img)
            data['slide_images'] = cleaned_images
            
        if 'roadmap' not in data or data['roadmap'] is None: data['roadmap'] = []
        if 'day_in_life_prompts' not in data or data['day_in_life_prompts'] is None: data['day_in_life_prompts'] = []

        client.table("saved_careers").upsert(data, on_conflict="user_id,career_uid").execute()
        return {"status": "success"}
    except Exception as e: 
        raise HTTPException(status_code=500, detail=str(e))

@app.post("/api/career-images")
async def upload_career_images(req: SaveCareerImageRequest, authorization: str = Header(None), user = Depends(verify_token)):
    if req.user_id != user.id: raise HTTPException(status_code=403)
    
    uploaded_urls = []
    
    async with httpx.AsyncClient(timeout=30.0) as http_client:
        for i, img_str in enumerate(req.images):
            if not img_str: 
                uploaded_urls.append(None)
                continue
                
            if img_str.startswith("http"):
                uploaded_urls.append(img_str)
                continue
                
            try:
                if "," in img_str: header, encoded = img_str.split(",", 1); ext = header.split(";")[0].split("/")[1]
                else: encoded = img_str; ext = "png"
                
                import base64
                file_data = base64.b64decode(encoded)
                file_path = f"{user.id}/{req.career_uid}/slide_{i}.{ext}"
                
                url = f"{SUPABASE_URL}/storage/v1/object/career_slideshows/{file_path}"
                
                resp = await http_client.post(
                    url,
                    content=file_data,
                    headers={
                        "Authorization": authorization,
                        "apikey": SUPABASE_KEY,
                        "Content-Type": f"image/{ext}",
                        "x-upsert": "true"
                    }
                )

                if resp.status_code == 200:
                    public_url = f"{SUPABASE_URL}/storage/v1/object/public/career_slideshows/{file_path}"
                    uploaded_urls.append(public_url)
                else:
                    logger.error(f"Raw HTTP Upload Failed: {resp.status_code} - {resp.text}")
                    uploaded_urls.append(None)

            except Exception as e: 
                uploaded_urls.append(None)

    return {"image_urls": uploaded_urls}
