def sanitize_input(text: str) -> str:
    if not isinstance(text, str): return str(text)
    return re.sub(r'[^\w\s.,!?-]', '', text).strip()

def sanitize_quiz_answers(answers: List[Dict[str, Any]]) -> str:
    clean_text = []
    for a in answers:
        q_id = a.get("questionId", "Unknown")
        ans = sanitize_input(str(a.get("answer", "")))
        clean_text.append(f"Q{q_id}: {ans}")
    return "; ".join(clean_text)

def get_user_client(token: str) -> Client:
    client = create_client(SUPABASE_URL, SUPABASE_KEY)
    client.postgrest.auth(token)
    return client

async def verify_token(authorization: str = Header(None)):
    if not authorization or not authorization.startswith("Bearer "):
        raise HTTPException(status_code=401, detail="Missing or invalid token")
    
    token = authorization.split(" ")[1]
    try:
        user = supabase.auth.get_user(token)
        if not user: raise HTTPException(status_code=401, detail="Invalid token")
        return user.user
    except Exception as e:
        logger.error(f"Auth error: {e}")
        raise HTTPException(status_code=401, detail="Authentication failed")

def increment_and_verify_quota(client: Client, user_id: str, count_col: str, date_col: str, limit: int, error_msg: str = None):
    try:
        res = client.table("profiles").select(f"{count_col}, {date_col}").eq("id", user_id).single().execute()
        if not res.data: raise HTTPException(status_code=404, detail="Profile not found")
        
        data = res.data
        count = data.get(count_col, 0)
        last_date_str = data.get(date_col)
        now = datetime.now(timezone.utc)

        if last_date_str:
            try:
                last_date = datetime.fromisoformat(last_date_str.replace('Z', '+00:00'))
            except ValueError: last_date = now
            
            if now.date() > last_date.date():
                client.table("profiles").update({count_col: 0, date_col: now.isoformat()}).eq("id", user_id).execute()
                count = 0
        
        if count >= limit:
            msg = error_msg or f"Daily limit of {limit} reached."
            raise HTTPException(status_code=429, detail=msg)
            
        client.rpc('increment_field', {'table_name': 'profiles', 'field_name': count_col, 'row_id': user_id, 'increment_val': 1}).execute()
        client.table("profiles").update({date_col: now.isoformat()}).eq("id", user_id).execute()

    except HTTPException as he: raise he
    except Exception as e:
        logger.error(f"Quota error: {e}")
        raise HTTPException(status_code=500, detail="Quota verification failed")

@app.post("/api/auth/signup")
async def signup(creds: UserLogin):
    try:
        res = supabase.auth.sign_up({"email": creds.email, "password": creds.password})
        if res.user and (res.user.identities is not None) and len(res.user.identities) == 0: raise Exception("User already registered")
        return res
    except Exception as e:
        msg = str(e)
        if "User already registered" in msg: raise HTTPException(status_code=400, detail="User already registered")
        raise HTTPException(status_code=400, detail=msg)

@app.get("/api/profile")
async def get_profile(authorization: str = Header(None), user = Depends(verify_token)):
    token = authorization.split(" ")[1]
    client = get_user_client(token)
    try:
        res = client.table("profiles").select("*").eq("id", user.id).single().execute()
        if not res.data: raise HTTPException(status_code=404, detail="Profile not found")
        profile = res.data
        profile['limits'] = {
            'daily_image_limit': DAILY_IMAGE_LIMIT,
            'daily_career_limit': DAILY_CAREER_LIMIT,
            'daily_general_quiz_limit': DAILY_GENERAL_QUIZ_LIMIT,
            'daily_details_view_limit': DAILY_DETAILS_VIEW_LIMIT,
            'slideshow_image_count': SLIDESHOW_IMAGE_COUNT
        }
        return profile
    except Exception: raise HTTPException(status_code=500, detail="Database error")

@app.post("/api/profile")
async def upsert_profile(profile: UserProfile, authorization: str = Header(None), user = Depends(verify_token)):
    if profile.id != user.id: raise HTTPException(status_code=403)
    token = authorization.split(" ")[1]
    client = get_user_client(token)
    try:
        data = profile.model_dump(by_alias=False)
        keys = [k for k in data if 'count' in k or 'date' in k or k == 'limits']
        for k in keys: del data[k]
        client.table("profiles").upsert(data).execute()
        return {"status": "success"}
    except Exception: raise HTTPException(status_code=500)


@app.post("/api/generate-domain")
async def generate_domain(req: GenerateDomainRequest, authorization: str = Header(None), user = Depends(verify_token)):
    token = authorization.split(" ")[1]
    client = get_user_client(token)
    increment_and_verify_quota(client, user.id, "daily_general_quiz_count", "last_career_generation_date", DAILY_GENERAL_QUIZ_LIMIT, "You have taken the general quiz too many times today.")
    try:
        prompt = f"Analyze quiz answers: {sanitize_quiz_answers(req.quiz_answers)}. Suggest domain: 'science', 'commerce', or 'arts'. Output JSON: {{ \"suggested_domain\": \"name\" }}"
        model = genai.GenerativeModel(TEXT_MODEL_NAME)
        response = await model.generate_content_async(prompt, generation_config={"response_mime_type": "application/json"})
        return json.loads(response.text)
    except Exception: return {"suggested_domain": "general"}

@app.post("/api/generate-career", response_model=GenerateCareerResponse)
async def generate_career(req: GenerateCareerRequest, authorization: str = Header(None), user = Depends(verify_token)):
    token = authorization.split(" ")[1]
    client = get_user_client(token)
    increment_and_verify_quota(client, user.id, "daily_career_generations_count", "last_career_generation_date", DAILY_CAREER_LIMIT)
    try:
        prompt = f"""
        Act as an expert Career Counselor.
        Profile: {req.user_profile.age}y/o, {req.user_profile.education_level} in {req.user_profile.specialization}.
        Residence: {req.user_profile.residence_country}, Target: {req.user_profile.preferred_work_country}.
        Traits: {sanitize_quiz_answers(req.quiz_answers)}.
        Task: Identify {CAREERS_PER_GENERATION} personalized career paths. Strict Eligibility (No Science for Arts students). Provide only high-level details (title, summary, salary, growth, tags). No roadmaps yet.
        OUTPUT FORMAT: STRICT JSON as {{ "recommendations": [ ... ] }}
        """
        model = genai.GenerativeModel(TEXT_MODEL_NAME)
        response = await model.generate_content_async(prompt, generation_config={"response_mime_type": "application/json"})
        result_json = json.loads(response.text)
        for rec in result_json.get("recommendations", []):
            if not rec.get("id"): rec["id"] = str(uuid.uuid4())
            rec["roadmap"] = []
            rec["dayInLifePrompts"] = []
        return {"recommendations": result_json["recommendations"]}
    except Exception as e:
        logger.error(f"Career Gen Error: {e}")
        raise HTTPException(status_code=500, detail="AI generation failed")

@app.post("/api/generate-images")
async def generate_images(req: GenerateImagesRequest, authorization: str = Header(None), user = Depends(verify_token)):
    token = authorization.split(" ")[1]
    client = get_user_client(token)
    increment_and_verify_quota(client, user.id, "daily_image_generations_count", "last_image_generation_date", DAILY_IMAGE_LIMIT)
    try:
        safe_title = sanitize_input(req.career_title)
        subject = f"a professional"
        if req.user_context: subject = f"a {req.future_age}yo {req.user_context.gender} {req.user_context.residence_country} national"
        async def gen_one(p):
            prompt = f"Cinematic shot of {subject} working as {safe_title}. Scene: {sanitize_input(p)}. Photorealistic 4k. Negative: text, words."
            for i in range(2):
                try:
                    model = genai.GenerativeModel(IMAGE_MODEL_NAME)
                    res = await model.generate_content_async(prompt)
                    if res.candidates[0].content.parts[0].inline_data:
                        return f"data:{res.candidates[0].content.parts[0].inline_data.mime_type};base64,{base64.b64encode(res.candidates[0].content.parts[0].inline_data.data).decode('utf-8')}"
                except: await asyncio.sleep(1.5)
            return None
        tasks = [gen_one(p) for p in req.prompts]
        return {"images": await asyncio.gather(*tasks)}
    except Exception: return {"images": [None]*len(req.prompts)}

@app.post("/api/saved-careers")

    token = authorization.split(" ")[1]
    client = get_user_client(token)
    try:
        data = career.model_dump(by_alias=False)
        data.update({"user_id": user.id, "career_uid": career.id})
        if 'id' in data: del data['id']
        if data.get('slide_images'): data['slide_images'] = [img if img and len(img) < 2000 else None for img in data['slide_images']]
        client.table("saved_careers").upsert(data, on_conflict="user_id,career_uid").execute()
        return {"status": "success"}
    except Exception as e: raise HTTPException(status_code=500, detail=str(e))

@app.post("/api/career-images")
async def upload_career_images(req: SaveCareerImageRequest, authorization: str = Header(None), user = Depends(verify_token)):
    if req.user_id != user.id: raise HTTPException(status_code=403)
    uploaded = []
    async with httpx.AsyncClient(timeout=30.0) as client:
        for i, img in enumerate(req.images):
            if not img or img.startswith("http"): 
                uploaded.append(img)
                continue
            try:
                ext = "png"
                if "," in img: ext = img.split(";")[0].split("/")[1]
                url = f"{SUPABASE_URL}/storage/v1/object/career_slideshows/{user.id}/{req.career_uid}/slide_{i}.{ext}"
                data = base64.b64decode(img.split(",")[1] if "," in img else img)
                await client.post(url, content=data, headers={"Authorization": authorization, "apikey": SUPABASE_KEY, "Content-Type": f"image/{ext}", "x-upsert": "true"})
                uploaded.append(f"{SUPABASE_URL}/storage/v1/object/public/career_slideshows/{user.id}/{req.career_uid}/slide_{i}.{ext}")
            except: uploaded.append(None)
    return {"image_urls": uploaded}
