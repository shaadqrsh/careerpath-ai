
import os
import json

import logging
import traceback
import base64
import urllib.parse
from datetime import datetime, timezone
from typing import List, Optional, Dict, Any

from fastapi import FastAPI, HTTPException, Depends, Header, Request
from fastapi.middleware.cors import CORSMiddleware

import google.generativeai as genai

# --- CONFIGURATION ----
SUPABASE_URL = os.environ.get("SUPABASE_URL")
SUPABASE_KEY = os.environ.get("SUPABASE_KEY")
GEMINI_API_KEY = os.environ.get("GEMINI_API_KEY")

# Constants
DAILY_GENERATION_LIMIT = 3
TEXT_MODEL_NAME = "gemini-2.5-flash"
IMAGE_MODEL_NAME = "gemini-2.5-flash-image" 

logging.basicConfig(level=logging.ERROR)
logger = logging.getLogger(__name__)

app = FastAPI()

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

if not SUPABASE_URL or not SUPABASE_KEY:
    logger.warning("Supabase credentials missing. Auth/DB will fail.")
    supabase: Client = None
else:
    supabase: Client = create_client(SUPABASE_URL, SUPABASE_KEY)

if not GEMINI_API_KEY:
    logger.warning("Gemini API Key missing. AI generation will fail.")
else:
    genai.configure(api_key=GEMINI_API_KEY)

@app.get("/")
async def health_check():
    return {"status": "ok", "service": "CareerPath AI Backend"}

# --- SCHEMAS ---

class BaseSchema(BaseModel):
    model_config = ConfigDict(
        alias_generator=to_camel,
        populate_by_name=True,
        from_attributes=True
    )

class UserLogin(BaseSchema):
    email: str
    password: str

class PasswordResetRequest(BaseSchema):
    email: str

class UpdatePasswordRequest(BaseSchema):
    password: str

class UserProfile(BaseSchema):
    id: str
    full_name: str
    gender: str
    age: int
    education_level: str
    specialization: str
    residence_country: str
    preferred_work_country: str

class CareerRoadmapStep(BaseSchema):
    title: str
    description: str
    local_path: Optional[str] = None
    target_path: Optional[str] = None
    duration: str

class CareerRecommendation(BaseSchema):
    id: str
    title: str
    match_score: int
    summary: str
    salary_range: str
    growth: str
    tags: List[str]
    is_pivot: Optional[bool] = False
    pivot_analysis: Optional[str] = None
    roadmap: List[CareerRoadmapStep]
    day_in_life_prompts: Optional[List[str]] = []
    slide_images: Optional[List[Optional[str]]] = []

class GenerateCareerRequest(BaseSchema):
    user_profile: UserProfile
    quiz_answers: List[Dict[str, Any]]

class GenerateImagesRequest(BaseSchema):
    career_title: str
    prompts: List[str]
    user_context: Optional[UserProfile] = None
    future_age: Optional[int] = 25

class SaveCareerImageRequest(BaseSchema):
    user_id: str
    career_uid: str
    images: List[Optional[str]]

class GenerateDomainRequest(BaseSchema):
    quiz_answers: List[Dict[str, Any]]

# --- DEPENDENCIES ---

async def verify_token(authorization: str = Header(None)):
    if not authorization or not authorization.startswith("Bearer "):
        raise HTTPException(status_code=401, detail="Missing or invalid token")
    
    token = authorization.split(" ")[1]
    
    try:
        user = supabase.auth.get_user(token)
        if not user:
            raise HTTPException(status_code=401, detail="Invalid token")
        return user.user
    except Exception as e:
        logger.error(f"Auth error: {e}")
        raise HTTPException(status_code=401, detail="Authentication failed")

# --- AUTH ROUTES ---

@app.post("/api/auth/signup")
async def signup(creds: UserLogin):
    try:
        res = supabase.auth.sign_up({
            "email": creds.email,
            "password": creds.password,
        })
        if res.user and (res.user.identities is not None) and len(res.user.identities) == 0:
            raise Exception("User already registered")
        return res
    except Exception as e:
        msg = str(e)
        if "User already registered" in msg:
             raise HTTPException(status_code=400, detail="User already registered")
        raise HTTPException(status_code=400, detail=msg)

@app.post("/api/auth/login")
async def login(creds: UserLogin):
    try:
        res = supabase.auth.sign_in_with_password({
            "email": creds.email,
            "password": creds.password,
        })
        return {
            "access_token": res.session.access_token,
            "user": res.user
        }
    except Exception as e:
        raise HTTPException(status_code=400, detail=str(e))

@app.post("/api/auth/reset-password")
async def reset_password(req: PasswordResetRequest):
    try:
        supabase.auth.reset_password_email(req.email)
        return {"status": "success", "message": "Password reset email sent"}
    except Exception as e:
        logger.error(f"Reset Password Error: {e}")
        raise HTTPException(status_code=500, detail="Failed to send reset email")

@app.post("/api/auth/update-password")
async def update_password(req: UpdatePasswordRequest, user = Depends(verify_token)):
    try:
        supabase.auth.update_user({ "password": req.password })
        return {"status": "success"}
    except Exception as e:
        logger.error(f"Update Password Error: {e}")
        msg = str(e)
        if "different from the old password" in msg:
             raise HTTPException(status_code=400, detail="New password must be different from the old password.")
        raise HTTPException(status_code=500, detail="Failed to update password")

@app.get("/api/auth/me")
async def get_current_user(user = Depends(verify_token)):
    return {"id": user.id, "email": user.email}

# --- PROFILE ROUTES ---

@app.get("/api/profile")
async def get_profile(user = Depends(verify_token)):
    try:
        res = supabase.table("profiles").select("*").eq("id", user.id).single().execute()
        if not res.data:
             raise HTTPException(status_code=404, detail="Profile not found")
        return res.data
    except Exception as e:
        # Check for PostgREST "0 rows" error (PGRST116) or the specific message
        err_str = str(e)
        if "PGRST116" in err_str or "The result contains 0 rows" in err_str or "JSON object requested, multiple (or no) rows returned" in err_str:
             raise HTTPException(status_code=404, detail="Profile not found")
        
        logger.error(f"Get Profile Error: {e}")
        raise HTTPException(status_code=500, detail="Database error")

@app.post("/api/profile")
async def upsert_profile(profile: UserProfile, user = Depends(verify_token)):
    if profile.id != user.id:
        raise HTTPException(status_code=403, detail="Cannot update other users")
    try:
        data = profile.model_dump(by_alias=False)
        # Exclude quota fields from direct update
        if 'daily_generations_count' in data: del data['daily_generations_count']
        if 'last_generation_date' in data: del data['last_generation_date']
        
        res = supabase.table("profiles").upsert(data).execute()
        return {"status": "success"}
    except Exception as e:
        logger.error(f"Upsert Profile Error: {e}")
        raise HTTPException(status_code=500, detail=str(e))

# --- AI GENERATION ROUTES ---

@app.post("/api/generate-domain")
async def generate_domain(req: GenerateDomainRequest, user = Depends(verify_token)):
    try:
        answers = req.quiz_answers
        prompt = f"""
        Analyze these quiz answers from a General Career Assessment.
        Identify the most suitable broad domain for this user.
        Answers: {json.dumps(answers, indent=2)}
        Options: 'science', 'commerce', 'arts'.
        Output strictly JSON: {{ "suggested_domain": "science" }}
        """
        model = genai.GenerativeModel(TEXT_MODEL_NAME)
        response = await model.generate_content_async(prompt, generation_config={"response_mime_type": "application/json"})
        return json.loads(response.text)
    except Exception as e:
        logger.error(f"Domain Gen Error: {e}")
        return {"suggested_domain": "general"}

@app.post("/api/generate-career")
async def generate_career(req: GenerateCareerRequest, user = Depends(verify_token)):
    try:
        user_p = req.user_profile
        answers = req.quiz_answers
        prompt = f"""
        Act as an expert Career Counselor using Global Market Data.
        Analyze this user profile: {user_p.full_name}, {user_p.age}y/o, {user_p.education_level} in {user_p.specialization}.
        Residence: {user_p.residence_country}, Target: {user_p.preferred_work_country}.
        Quiz Context: {json.dumps(answers, indent=2)}.
        
        Task:
        1. Identify 5 personalized career paths based on the profile and quiz.
        2. STRICT ELIGIBILITY CHECK: If user is Arts, no Science. If Pivot, verify realistic paths.
        3. PIVOT ANALYSIS: Is this a "Natural Progression" or "Major Pivot"?
        4. ROADMAP: Split into 'localPath' ({user_p.residence_country}) and 'targetPath' ({user_p.preferred_work_country}) if different.
        5. DATA INTEGRITY: Ensure 'salaryRange', 'growth', and 'roadmap' are FILLED for every recommendation. Do not leave null.
        
        OUTPUT FORMAT: STRICT JSON.
        JSON Structure:
        {{
          "recommendations": [
            {{
              "id": "uuid-placeholder",
              "title": "Job Title",
              "matchScore": 85,
              "summary": "Full detailed summary...",
              "salaryRange": "$50k - $80k",
              "growth": "High (20% by 2030)",
              "tags": ["Tag1", "Tag2"],
              "isPivot": boolean,
              "pivotAnalysis": "Analysis string...",
              "roadmap": [
                 {{ "title": "Step 1", "description": "...", "localPath": "...", "targetPath": "...", "duration": "1 Year" }},
                 {{ "title": "Step 2", "description": "...", "localPath": "...", "targetPath": "...", "duration": "2 Years" }},
                 {{ "title": "Step 3", "description": "...", "localPath": "...", "targetPath": "...", "duration": "1 Year" }}
              ],
              "dayInLifePrompts": ["Scene 1 description", "Scene 2 description", "Scene 3 description"]
            }}
          ]
        }}
        """
        model = genai.GenerativeModel(TEXT_MODEL_NAME)
        response = await model.generate_content_async(prompt, generation_config={"response_mime_type": "application/json"})
        result_json = json.loads(response.text)
        for rec in result_json.get("recommendations", []):
            if not rec.get("id"): rec["id"] = str(uuid.uuid4())
        return result_json
    except Exception as e:
        logger.error(f"Career Gen Error: {e}")
        raise HTTPException(status_code=500, detail="AI generation failed")

@app.post("/api/generate-images")
async def generate_images(req: GenerateImagesRequest, user = Depends(verify_token)):
    # 1. QUOTA CHECK
    current_count = 0
    try:
        profile_res = supabase.table("profiles").select("daily_generations_count, last_generation_date").eq("id", user.id).single().execute()
        if profile_res.data:
            profile_data = profile_res.data
            last_date_str = profile_data.get('last_generation_date')
            count = profile_data.get('daily_generations_count', 0)
            
            now = datetime.now(timezone.utc)
            last_date = datetime.fromisoformat(last_date_str.replace('Z', '+00:00')) if last_date_str else now
            
            # Reset if it's a new day
            if (now - last_date).days >= 1 or now.date() > last_date.date():
                count = 0
            
            if count >= DAILY_GENERATION_LIMIT:
                raise HTTPException(status_code=429, detail=f"Daily visualization limit reached ({DAILY_GENERATION_LIMIT}/day). Please try again tomorrow.")
            
            current_count = count
    except HTTPException as he:
        raise he
    except Exception as e:
        logger.error(f"Quota check failed: {e}")
        # Fail safe: if we can't check quota, we block to prevent abuse, or you could allow. Blocking for safety.
        raise HTTPException(status_code=500, detail="Unable to verify generation quota.")

    # 2. GENERATION
    try:
        subject_desc = "a professional"
        if req.user_context:
            u = req.user_context
            age = req.future_age or (u.age + 5)
            subject_desc = f"a {age} year old {u.gender} {u.residence_country} national"

        import asyncio
        async def generate_single_image(prompt_text):
            full_prompt = f"Cinematic shot of {subject_desc} working as a {req.career_title}. Scene: {prompt_text}. Style: Photorealistic, 4k, dramatic lighting."
            
            # --- 1. ATTEMPT GEMINI (IMAGEN) ---
            try:
                # Use Gemini 2.5 Flash Image logic (generate_content with prompt)
                model = genai.GenerativeModel(IMAGE_MODEL_NAME)
                response = await model.generate_content_async(full_prompt)
                
                # Check for inline data (image)
                if response.candidates and response.candidates[0].content.parts:
                    for part in response.candidates[0].content.parts:
                        if part.inline_data:
                            # part.inline_data.data is bytes. Must encode to base64 string.
                            b64_data = base64.b64encode(part.inline_data.data).decode('utf-8')
                            return f"data:{part.inline_data.mime_type};base64,{b64_data}"
            except Exception as e:
                print("--- GEMINI GENERATION ERROR ---")
                print(f"Prompt: {full_prompt}")
                traceback.print_exc()
                logger.warning(f"Gemini Image Gen Failed: {e}")

            # --- 2. FALLBACK: POLLINATIONS.AI ---
            try:
                safe_prompt = full_prompt[:400] 
                encoded_prompt = urllib.parse.quote(safe_prompt)
                seed = uuid.uuid4().int % 10000
                image_url = f"https://image.pollinations.ai/prompt/{encoded_prompt}?width=1280&height=720&seed={seed}&nologo=true"
                
                headers = {'User-Agent': 'CareerPathAI/1.0'}
                async with httpx.AsyncClient(follow_redirects=True, timeout=30.0) as client:
                    resp = await client.get(image_url, headers=headers)
                    if resp.status_code == 200:
                        b64_img = base64.b64encode(resp.content).decode('utf-8')
                        return f"data:image/jpeg;base64,{b64_img}"
                    else:
                         print(f"Pollinations Error Status: {resp.status_code}")
            except Exception as e:
                print("--- POLLINATIONS ERROR ---")
                traceback.print_exc()
                logger.error(f"Pollinations Exception: {str(e)}")

            # --- 3. FAIL (Return None) ---
            return None

        tasks = [generate_single_image(p) for p in req.prompts]
        generated_images = await asyncio.gather(*tasks)
        
        # 3. UPDATE QUOTA (Only if at least one image generated)
        if any(img is not None for img in generated_images):
            try:
                supabase.table("profiles").update({
                    "daily_generations_count": current_count + 1,
                    "last_generation_date": datetime.now(timezone.utc).isoformat()
                }).eq("id", user.id).execute()
            except Exception as e:
                logger.error(f"Failed to update quota: {e}")

        return {"images": generated_images}
        
    except Exception as e:
        logger.error(f"Image API Global Error: {e}")
        # Return all Nones to signify failure, UI will handle
        return {"images": [None for _ in req.prompts]}

@app.get("/api/saved-careers")
async def get_saved_careers(user = Depends(verify_token)):
    try:
        res = supabase.table("saved_careers").select("*").eq("user_id", user.id).execute()
        return res.data
    except Exception: return []

@app.post("/api/saved-careers")
async def save_career(career: CareerRecommendation, user = Depends(verify_token)):
    try:
        data = career.model_dump(by_alias=False)
        data['user_id'] = user.id
        data['career_uid'] = career.id
        if 'id' in data: del data['id']
        supabase.table("saved_careers").upsert(data, on_conflict="user_id,career_uid").execute()
        return {"status": "success"}
    except Exception as e: raise HTTPException(status_code=500, detail=str(e))

@app.delete("/api/saved-careers/{career_uid}")
async def delete_career(career_uid: str, user = Depends(verify_token)):
    try:
        # 1. Delete DB Record
        supabase.table("saved_careers").delete().eq("user_id", user.id).eq("career_uid", career_uid).execute()
        
        # 2. Cleanup Storage
        try:
            files_to_remove = []
            for i in range(10): # Cleanup up to 10 slides
                files_to_remove.append(f"{user.id}/{career_uid}/slide_{i}.png")
                files_to_remove.append(f"{user.id}/{career_uid}/slide_{i}.jpeg")
                files_to_remove.append(f"{user.id}/{career_uid}/slide_{i}.jpg")
            
            supabase.storage.from_("career_slideshows").remove(files_to_remove)
        except Exception as storage_e:
            logger.warning(f"Storage cleanup failed (non-critical): {storage_e}")

        return {"status": "success"}
    except Exception as e: raise HTTPException(status_code=500, detail=str(e))

@app.post("/api/career-images")
async def upload_career_images(req: SaveCareerImageRequest, user = Depends(verify_token)):
    if req.user_id != user.id: raise HTTPException(status_code=403)
    uploaded_urls = []
    for i, img_str in enumerate(req.images):
        if not img_str: # Handle None/Empty
            uploaded_urls.append(None)
            continue
            
        if img_str.startswith("http"):
            uploaded_urls.append(img_str)
            continue
            
        try:
            if "," in img_str: header, encoded = img_str.split(",", 1); ext = header.split(";")[0].split("/")[1]
            else: encoded = img_str; ext = "png"
            import base64
            file_data = base64.b64decode(encoded)
            file_path = f"{user.id}/{req.career_uid}/slide_{i}.{ext}"
            supabase.storage.from_("career_slideshows").upload(file_path, file_data, {"content-type": f"image/{ext}", "upsert": "true"})
            public_url = supabase.storage.from_("career_slideshows").get_public_url(file_path)
            uploaded_urls.append(public_url)
        except Exception: uploaded_urls.append(img_str)
    return {"image_urls": uploaded_urls}
